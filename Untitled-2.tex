\begin{equation}
V_{out} = V_{in} \cdot \frac{R_{th}}{R_{th} + R_1}
\end{equation}

\begin{equation}
ADC = \frac{V_{out}}{V_{in}} \cdot 1023
\end{equation}

\begin{equation}
ADC = \frac{R_{th}}{R_{th} + R_1} \cdot 1023
\end{equation}

\begin{equation}
R_{th} = R_1 \cdot \frac{1023}{ADC} - R_1 = \frac{R_1}{(\frac{1023}{ADC}) - 1}
\end{equation}
